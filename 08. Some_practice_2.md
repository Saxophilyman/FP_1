```python
from functools import reduce

def second_max(numbers: list[int]) -> int:

  first, second = numbers[0], numbers[1]

  def step(acc: tuple[int, int], value : int) -> tuple[int, int]:
```

```java
import java.util.Arrays;

public class SecondMaximum {

    private record MaxPair(int maximum, int secondMaximum) {

        static MaxPair of(int first, int second) {
            return new MaxPair(
                    Math.max(first, second),
                    Math.min(first, second)
            );
        }

        MaxPair add(int value) {
            if (value >= maximum) {
                return new MaxPair(value, maximum);
            }

            if (value > secondMaximum) {
                return new MaxPair(maximum, value);
            }

            return this;
        }

        MaxPair combine(MaxPair other) {
            return add(other.maximum)
                    .add(other.secondMaximum);
        }
    }

    public static int secondMax(int[] numbers) {
        if (numbers.length < 2) {
            throw new IllegalArgumentException(
                    "Нужно минимум два числа"
            );
        }

        MaxPair initial = MaxPair.of(numbers[0], numbers[1]);

        MaxPair result = Arrays.stream(numbers, 2, numbers.length)
                .boxed()
                .reduce(initial, MaxPair::add, MaxPair::combine);

        return result.secondMaximum();
    }
```
