from pymonad.state import State

def parse_battalion(battalion):
    if not battalion:
        return set()

    x, y, *rest = battalion
    return {(x, y)} | parse_battalion(rest)

def neighbours(cell, n, m):
    x, y = cell

    possible = {
        (x + 1, y),
        (x - 1, y),
        (x, y + 1),
        (x, y - 1)
    }

    return {
        (a, b)
        for a, b in possible
        if 1 <= a <= n and 1 <= b <= m
    }

def expand(captured, n, m):
    border = set().union(
        *(neighbours(cell, n, m) for cell in captured)
    )

    return captured | border

def one_day(n, m):
    def operation(captured):
        def computation(day):
            new_captured = expand(captured, n, m)
            return new_captured, day + 1

        return State(computation)

    return operation

def all_captured(captured, n, m):
    return len(captured) == n * m


def simulate(n, m, captured_state):
    captured, day = captured_state.run(1)

    if all_captured(captured, n, m):
        return day

    return simulate(n, m, captured_state.then(one_day(n, m)))

def conquest_campaign(n, m, l, battalion):
    initial_captured = parse_battalion(battalion)

    return simulate(n, m, State.insert(initial_captured))


print(conquest_campaign(3, 4, 2, [2, 2, 3, 4]))
