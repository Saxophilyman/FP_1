```python
from pymonade.maybe import Just, Nothing

def to_left(num):
  def step(pole):
    left, right = pole
    new_left = left + num

    if abs(new_left - right) > 4
      return Nothing

    return Just([new_left, right])
  return step


def to_right(num):
  def step(pole):
    left, right = pole
    new_right = right + num

    if abs(new_right - left) > 4:
      return Nothing

    return Just([left, new_right])
  return step

def banana(pole):
```
