```python
def odometer(oksana: list[int]) -> int:
    speeds = oksana[::2]
    times = oksana[1::2]

    previous_times = [0] + times[:-1]

    distances = (
        speed * (time - previous_time)
        for speed, time, previous_time
        in zip(speeds, times, previous_times)
    )

    total_distance = sum(distances)

    return total_distance
```
