```python
from pymonad.maybe import Just
from pymonad.list import ListMonad

@curry(2)
def add(x, y):
    return x + y

def add10(value):
    return value.map(add(10))

print(add10(Just(5)))
print(add10(ListMonad(1, 2, 3)))
```

3.2.  
Гораздо заковырестей, т.к. надо было выяснить фишки питона в работе со словарём и вытаскиванием значений

```python
from pymonad.tools import curry

@curry(3)
def tag(tag_name: str, attr: dict[str, str], content: str) -> str:
    attributes = " ".join(
        f'{key}="{attribute}"'
        for key, attribute in attr.items()
    )

    if attributes:
        return f"<{tag_name} {attributes}>{content}</{tag_name}>"

    return f"<{tag_name}>{content}</{tag_name}>"
```
